        private static int Ip3ToInt(string ip4)
        {
            var arr = ip4.Split('.');
            var p0 = int.Parse(arr[0]);
            var p1 = int.Parse(arr[1]);
            var p2 = int.Parse(arr[2]);

            return (p0 << 16) | (p1 << 8) | p2;
        }

        private static string IntToIp3(int s)
        {
            var p0 = s >> 16 & 0xFF;
            var p1 = s >> 8 & 0xFF;
            var p2 = s & 0xFF;
            return string.Format("{0}.{1}.{2}.0", p0, p1, p2);
        }
